```java
@Data
@Entity
@NoArgsConstructor
@Table(name = "Item")
public class Item {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    @NotNull
    @Column(name = "name", unique = false)
    private String name;

    @NotNull
    @Column(name = "number")
    private String number;

    private int region;

    @Builder
    public Item(String name, String number, int region){
        this.name = name;
        this.number = number;
        this.region = region;
    }
}
```
